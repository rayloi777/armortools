Unity-style 多層 Culling 系統

GPU Frustum → LOD → Baked Occlusion

Culling Pipeline（執行順序）

1
GPU Frustum Culling — ComputeShader
所有 RenderObject3D 的 AABB 與 camera frustum 6 planes 做 test
輸出：visible object list (Buffer)
2
LOD Selection — CPU + GPU
根據 camera 距離選擇 mesh LOD level
每個物件根據 distance bucket 選擇：LOD0 / LOD1 / LOD2 / LOD3 / ShadowOnly
3
Baked Occlusion Culling — Pre-compute + Runtime query
Editor: 烘焙 occlusion data (Cell Grid + PVS)
Runtime: 根據 camera cell 查詢 visible list
4
Draw Call — DrawProceduralIndirect
只對最終 visible objects 發送 draw call

GPU Frustum Culling ComputeShader

// Object Data Buffer (輸入) struct ObjectBounds { float3 center; // AABB center (world space) float3 extents; // AABB half-extents uint lodMask; // 該物件支援哪些 LOD level }; // Frustum Plane (輸入) struct Plane { float3 n; float d; }; // dot(n,p) + d = 0 Plane frustum[6]; // near/far/left/right/top/bottom // Output: visible objects (append to buffer) [numthreads(256, 1, 1)] void FrustumCullKernel( uint id : SV_DispatchThreadID, ObjectBounds obj : ObjectBounds, Plane planes[6] : FRUSTUM_PLANES, RWStructuredBuffer visibleOut : VISIBLE_OUT ) { // 對 6 個 plane 做 AABB outside test bool culled = false; for (int i = 0; i < 6; i++) { float d = dot(obj.center, planes[i].n) + planes[i].d; if (d < -length(obj.extents)) { culled = true; break; } } if (!culled) { uint index = outBuffer.IncrementCounter(); visibleOut[index] = id; } }

LOD System

// comp_lod 組件 typedef struct { float distances[4]; // LOD0→LOD1, LOD1→LOD2, ... char* mesh_lod0_path; // 最高解析度 char* mesh_lod1_path; // 次高 char* mesh_lod2_path; // 中等 char* mesh_lod3_path; // 最低 int current_lod; // 運行時選擇的 LOD } comp_lod; // 選擇 LOD int select_lod(float dist) { if (dist > distances[2]) return 3; if (dist > distances[1]) return 2; if (dist > distances[0]) return 1; return 0; }

Occlusion Culling（烘焙版）

// Cell Grid 分區 struct OcclusionCell { float3 min; // cell 邊界 float3 max; uint visibleObjects[MAX_VIS]; // 預計算可見列表 uint numVisible; }; OcclusionCell cells[GRID_W * GRID_H * GRID_D]; // Runtime: 根據 camera 位置找 cell OcclusionCell* cell = get_cell_at(cameraPos); for (i = 0; i < cell.numVisible; i++) { add_to_visible_list(cell.visibleObjects[i]); }
