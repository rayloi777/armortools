Directional Light

typedef struct { float dir_x, dir_y, dir_z; // 光照方向（歸一化） float color_r, g, b; float strength; bool enabled; // CSM bool cast_shadows; float shadow_bias; float normal_offset_bias; float shadow_radius; // PCF kernel 大小 } comp_directional_light;

Point Light

typedef struct { float pos_x, pos_y, pos_z; // 世界座標 float color_r, g, b; float strength; float range; // 衰減半徑 bool enabled; // Shadow bool cast_shadows; float shadow_bias; // Cubemap stored as: shadow_cube_0 .. shadow_cube_7 } comp_point_light;

Spot Light

typedef struct { float pos_x, pos_y, pos_z; // 世界座標 float dir_x, dir_y, dir_z; // 照射方向 float color_r, g, b; float strength; float range; // 衰減半徑 float inner_cone; // 內圓錐角（rad） float outer_cone; // 外圓錐角（rad） bool enabled; bool cast_shadows; float shadow_bias; } comp_spot_light;

Deferred Lighting Shader（deferred_light.kong 擴展）

// 燈光 Buffer（所有燈光資料） struct LightData { // Directional[4] float4 dir[4]; // xyz = direction, w = enabled float4 dirColor[4]; float4 dirSMVP[4*3]; // 4 lights × 3 cascades float4 dirCascadeSplit[4]; // Point[8] float4 pointPos[8]; float4 pointColor[8]; float pointRange[8]; bool pointShadow[8]; // Spot[4] float4 spotPos[4]; float4 spotDir[4]; float4 spotColor[4]; float spotRange[4]; float spotInnerCone[4]; float spotOuterCone[4]; }; fun lighting_fragment(input : vert_out) : float4 { float4 g0 = sample(gbuffer0, input.tex); float4 g1 = sample(gbuffer1, input.tex); float depth = sample(gbufferD, input.tex).r; float3 N = decode_octahedron(g0.xy); float3 albedo = g1.rgb; float metallic = g0.a, roughness = g0.b; float ao = g1.a; float3 color = ambient * albedo; // --- Directional Lights --- for (int i = 0; i < 4; i++) { if (!dir[i].w) continue; float3 L = -normalize(dir[i].xyz); float NdotL = max(0, dot(N, L)); float shadow = sample_csm(L, input.pos, dirSMVP[i], dirCascadeSplit[i]); color += NdotL * shadow * dirColor[i].rgb * albedo; } // --- Point Lights --- for (int i = 0; i < 8; i++) { if (!pointPos[i].w) continue; // enabled flag float3 toLight = pointPos[i].xyz - worldPos; float dist = length(toLight); if (dist > pointRange[i]) continue; float3 L = toLight / dist; float att = 1.0 - dist / pointRange[i]; att = att * att; float NdotL = max(0, dot(N, L)); float shadow = pointShadow[i] ? sample_cubemap(i, L, worldPos) : 1.0; color += NdotL * att * shadow * pointColor[i].rgb * albedo; } // --- Spot Lights --- for (int i = 0; i < 4; i++) { if (!spotPos[i].w) continue; float3 toLight = spotPos[i].xyz - worldPos; float dist = length(toLight); if (dist > spotRange[i]) continue; float3 L = toLight / dist; float spotEffect = dot(L, -normalize(spotDir[i].xyz)); float angleAtten = smoothstep(spotOuterCone[i], spotInnerCone[i], spotEffect); float att = (1.0 - dist / spotRange[i]) * angleAtten; float NdotL = max(0, dot(N, L)); float shadow = sample_spot_shadow(i, worldPos); color += NdotL * att * shadow * spotColor[i].rgb * albedo; } return float4(color, 1.0); }

光照衰減公式

Point Light

att = 1.0 - saturate(dist / range) att² = att * att // 二次衰減更物理精確

Spot Light

spotEffect = dot(L, -spotDir) angleAtt = smoothstep(outer, inner, spotEffect) totalAtt = (1 - dist/range)² × angleAtt
